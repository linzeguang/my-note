### 算数运算符

```dart
// + - * / % 加减乘除取余
print(1 + 2); // 3
print(1 - 2); // -1
print(2 * 3); // 6
print(5 / 2); // 2.5
// ~/ 除，结果取整，是向下取整
print(5 ~/ 2); // 2

// ++ --
num a = 1
num b = ++a // 赋值前自增a，b = 2， a = 2

a = 1
num c = a++ // 赋值后自增a，b = 1, a = 2

// -- 逻辑相同

// == != > < >= <= 没有绝对等于 '==='

// 类型测试操作符
// as 用于类型转换
// is is! 用于类型判断
print(b is num); // true
print(b is String); // false
print(b is! String); // true

// 赋值操作符
// = -= += /= %= *= ~/= 了解

// ??=   当b为空的时候，则将值赋给b，否则，b保持不变
num a = 1;
num b;
b ??= a; // b = 1
num c = 2
c ??= a // c = 2
```
